export const SplitButtonControlled = ({ children, ...restProps }) => {
    const [selectedIndex, setSelectedIndex] = React.useState(0);
    return (
        <SplitButton {...restProps} selectedIndex={selectedIndex} onSelectIndex={(index) => setSelectedIndex(index)}>
            {children}
        </SplitButton>
    );
};
